\begin{defi}{}{}
Si la probabilidad $P_{ij} > 0$ para algún $n \geq 1$ entonces se dice que $i$ alcanza a $j$. Se denota por $i \to j$.
\end{defi}

\begin{defi}{}{}
    Dos estados se comunican si $i \to j$ y $j \to i$. Se denota por $i \leftrightarrow j$.
\end{defi}

\begin{defi}{}{}
    Una cadena de Markov es irreducible si todos los estados se comunican entre sí.
\end{defi}

\begin{defi}{}{}
    Si $j \leftrightarrow i$ se dice que $j$ es un estado de retorno   
\end{defi}

\begin{defi}{}{}
    Una clase de comunicación $E_{j}$ es el conjunto de todos los estados que $j \leftrightarrow k$, $j .k \in E_{j}$
\end{defi}

\begin{teo}{}{}
    Si $E_{1}$ y $E_{2}$ son dos clases de comunicación, entonces $E_{1} = E_{2}$ si $E_{1} \cap E_{2} \neq \emptyset$.
\end{teo}

\begin{nota}[Importante]
    Una Cadena de Markov se puede dividir en clases de comunicación o de estados de no retorno.
\end{nota}

\end{defi}
