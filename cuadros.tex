\newtcolorbox[auto counter,number within=section]{mybox}[2][]{%
  enhanced,
  skin=enhancedlast jigsaw,
  attach boxed title to top left={xshift=-4mm,yshift=-0.5mm},
  fonttitle=\bfseries\sffamily\color{black},
  varwidth boxed title=0.7\linewidth,
  colbacktitle=green!45!white,
  colframe=cyan!50!black,
  interior style={top color=green!10!white,bottom color=cyan!10!white},
  boxed title style={empty,arc=0pt,outer arc=0pt,boxrule=0pt},
  underlay boxed title={
    \fill[green!45!white] (title.north west)-- (title.north east)-- +(\tcboxedtitleheight-1mm,-\tcboxedtitleheight+1mm)-- ([xshift=4mm,yshift=0.5mm]frame.north east)-- +(0mm,-1mm)-- (title.south west)-- cycle;
    \fill[green!45!white!50!black] ([yshift=-0.5mm]frame.north west)-- +(-0.4,0)-- +(0,-0.3)-- cycle;
    \fill[green!45!white!50!black] ([yshift=-0.5mm]frame.north east)-- +(0,-0.3)-- +(0.4,0)-- cycle;
  },
  before upper={\refstepcounter{example}\label{#1}},
  title=Ejercicio~\thetcbcounter: #2,#1
}
